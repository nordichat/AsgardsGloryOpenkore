automacro auto_loot_activation {
    run-once 1
    exclusive 1
    call enable_autoloot
}

macro enable_autoloot {
    do c @autoloot 100
}
automacro ifInventoryContainsItem1 {
    exclusive 1
    run-once 1
    timeout 5
    inventory "Rough Oridecon" > 0
    call transferItem
}

automacro ifInventoryContainsItem2 {
    exclusive 1
    run-once 1
    timeout 5
    inventory "Creamy Card" > 0
    call transferItem
}

automacro ifInventoryContainsItem3 {
    exclusive 1
    run-once 1
    timeout 5
    inventory "Flora Card" > 0
    call transferItem
}

automacro ifInventoryContainsItem4 {
    exclusive 1
    run-once 1
    timeout 5
    inventory "Earth Petite Card" > 0
    call transferItem
}

macro transferItem {
    $mainCharname = "Stalker1"
    $charname = @eval($::char->{'name'})

    if ($charname != $mainCharname) {
        do ai manual
        $item1 = @inventory(Rough Oridecon)
        if ($item1 > -1) {
            do drop $item1
            do take first
        }

        $item2 = @Inventory(Creamy Card)
        if ($item2 > -1) {  
            do drop $item2
            do take first
        }
        
        $item3 = @Inventory(Flora Card)
        if ($item3 > -1) {  
            do drop $item3
            do take first
        }
        
        $item4 = @Inventory(Earth Petite Card)
        if ($item4 > -1) {  
            do drop $item4
            do take first
        }

        do ai auto
        release ifInventoryContainsItem1
        release ifInventoryContainsItem2
        release ifInventoryContainsItem3
        release ifInventoryContainsItem4
    } 
}

automacro monsterRefresh {
    monster /NPC/, /NPC Unknown/
    call {
        do c @refresh
        do e hmm
        do p "Refreshing due to unexpected issue."
    }
    timeout 3
}

automacro healAndReturn {
    set macro_delay 0.2
    hp < 15%
    call {
        do p @go geffen
        pause 1
        do e omg
        pause 1
        do move 120 70
        pause 1
        do talk @npc(/Healer/)
        do e lv2
        $mainCharname = "Stalker1"
        $charname = @eval($::char->{'name'})
        if ($charname == $mainCharname) {
            do p "Healed and ready to continue."
        }
    }
    timeout 60
}
