# This automacro initializes the loader and calls the main controller macro.
automacro InitializeLoader {
    run-once 1
    exclusive 1
    call MainController
}

# This macro serves as the main controller and calls the macro to configure basic settings.
macro MainController {
    call ConfigureBasicSettings
}

# This automacro configures the character for leveling when it has the 'First Aid' skill and is below job level 10.
macro WhatJobAndId {
    $jobId = @eval ($::char->{'jobID'})
    $jobName = @eval ($::jobs_lut{$jobId})

    log $jobName $jobId 

    do c $jobName $jobId
}
